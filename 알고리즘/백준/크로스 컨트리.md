import java.io.*;
import java.util.*;

public class Main {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        int T = Integer.parseInt(br.readLine());
        StringTokenizer st;

        for(int i = 0; i < T; i++) {
            int N = Integer.parseInt(br.readLine());
            st = new StringTokenizer(br.readLine());
            int[] runner = new int[N];
            for(int j = 0; j < N; j++) {
                runner[j] =  Integer.parseInt(st.nextToken());
            }

            Map<Integer, Integer> map = new HashMap<>();
            for(int r : runner) {
                map.put(r, map.getOrDefault(r, 0) + 1);
            }

            int s = 1;
            int[] scores = new int[N];
            for(int j = 0; j < N; j++) {
                if(map.get(runner[j]) != 6) continue;
                scores[j] = s;
                s++;
            }

            int answerSum = Integer.MAX_VALUE;
            int answerFive = Integer.MAX_VALUE;
            int answerTeam = 0;
            for(int key : map.keySet()) {
                if(map.get(key) != 6) continue;
                int sum = 0;
                int five = 0;
                int cnt = 0;
                for(int k = 0; k < N; k++) {
                    if(runner[k] == key && cnt == 4) {
                        five = scores[k];
                        break;
                    }
                    if(runner[k] == key && cnt < 4) {
                        sum += scores[k];
                        cnt++;
                    }
                }

                if(answerSum > sum || (answerSum == sum && answerFive > five)) {
                    answerSum = sum;
                    answerTeam = key;
                    answerFive = five;
                }
            }
            System.out.println(answerTeam);
        }
    }
}
