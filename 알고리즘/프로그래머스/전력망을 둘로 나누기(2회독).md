import java.util.*;

class Solution {
    public int solution(int n, int[][] wires) {
        int answer = n;
        int[][] map = new int[n + 1][n + 1];
        for(int[] w : wires) {
            map[w[0]][w[1]] = 1;
            map[w[1]][w[0]] = 1;
        }
        
        for(int i = 1; i <= n; i++) {
            for(int j = 1; j <= n; j++) {
                if(map[i][j] == 1) {
                    map[i][j] = 0;
                    map[j][i] = 0;
                    answer = Math.min(answer, calculate(map, i, n));
                    map[i][j] = 1;
                    map[j][i] = 1;
                }
            }
        }
        
        return answer;
    }
    
    private int calculate(int[][] map, int x, int n) {
        Queue<Integer> queue = new LinkedList<>();
        queue.add(x);
        
        int count = 1;
        boolean[] visited = new boolean[n + 1];
        visited[x] = true;
        
        while(!queue.isEmpty()) {
            int poll = queue.poll();
            for(int i = 1; i <= n; i++) {
                if(map[poll][i] == 1 && !visited[i]) {
                    visited[i] = true;
                    count++;
                    queue.add(i);
                }
            }
        }
        
        return Math.abs(n - (2 * count));
    }
}
