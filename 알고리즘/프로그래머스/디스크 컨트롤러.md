import java.util.*;

class Solution {
    public int solution(int[][] jobs) {
        int answer = 0;
        PriorityQueue<int[]> pq = new PriorityQueue<>((o1, o2) -> {
            if(o1[1] == o2[1]) {
                if(o1[0] == o2[0]) return o1[2] - o2[2];
                else return o1[0] - o2[0];
            } else return o1[1] - o2[1];
        });
        
        PriorityQueue<int[]> job = new PriorityQueue<>((o1, o2) -> o1[0] - o2[0]);
        
        for(int i = 0; i < jobs.length; i++) {
            if(jobs[i][0] == 0) pq.add(new int[] {jobs[i][0], jobs[i][1], i});
            else job.add(new int[] {jobs[i][0], jobs[i][1], i});
        }
        
        int[] returnTime = new int[jobs.length];
        int index = 0;
        int currentTime = 0;
       
        while(!pq.isEmpty() || !job.isEmpty()) {
            if(!pq.isEmpty()) {
                int[] poll = pq.poll();
                returnTime[index++] = currentTime + poll[1] - poll[0];
                currentTime += poll[1];
            } else {
                currentTime = job.peek()[0];
            }
            
            while(!job.isEmpty() && job.peek()[0] <= currentTime) {
                pq.add(job.poll());
            }
        }
        
        for(int t : returnTime) {
            answer += t;
        }
        
        answer /= returnTime.length;
        return answer;
    }
}
