import java.util.*;

class Solution {
    private PriorityQueue<String> pq = new PriorityQueue<>();
    private boolean[] visited;
    public String[] solution(String[][] tickets) {
        visited = new boolean[tickets.length];
        DFS(tickets, "ICN", "ICN", 0);
        
        return pq.poll().split(" ");
    }
    private void DFS(String[][] tickets, String route, String dep, int cnt) {
        if(cnt == tickets.length) {
            pq.add(route);
            return;
        }
        
        for(int i = 0; i < tickets.length; i++) {
            if(!visited[i] && tickets[i][0].equals(dep)) {
                visited[i] = true;
                DFS(tickets, route + " " + tickets[i][1], tickets[i][1], cnt + 1);
                visited[i] = false;
            }
        }
    }
}
