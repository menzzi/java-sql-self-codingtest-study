import java.util.*;

class Solution {
    private int[][] map;
    private boolean[][] visited;
    
    private final int[] dx = {1, 0, -1, 0};
    private final int[] dy = {0, 1, 0, -1};
    
    public int solution(int[][] rectangle, int characterX, int characterY, int itemX, int itemY) {
        map = new int[101][101];
        visited = new boolean[101][101];
        
        for(int[] rec : rectangle) {
            makeMap(rec[0] * 2, rec[1] * 2, rec[2] * 2, rec[3] * 2);
        }
        
        Queue<int[]> queue = new LinkedList<>();
        queue.add(new int[]{characterX * 2, characterY * 2, 0});
        
        while(!queue.isEmpty()) {
            int[] poll = queue.poll();
            if(poll[0] == itemX * 2 && poll[1] == itemY * 2) return poll[2]/2;
            
            for(int i = 0; i < 4; i++) {
                int nx = poll[0] + dx[i];
                int ny = poll[1] + dy[i];
                
                if(nx < 0 || ny < 0 || nx > 100 || ny > 100) continue;
                
                if(!visited[ny][nx] && map[ny][nx] == 2) {
                    visited[ny][nx] = true;
                    queue.add(new int[] {nx, ny, poll[2] + 1});
                }
            }
        }
        
        return -1;
    }
    private void makeMap(int x1, int y1, int x2, int y2) {
        for(int i = y1; i <= y2; i++) {
            for(int j = x1; j <= x2; j++) {
                if(map[i][j] == 1) continue;
                if(i == y1 || i == y2 || j == x1 || j == x2) {
                    map[i][j] = 2;
                    continue;
                }
                map[i][j] = 1;
            }
        }
    }
}
