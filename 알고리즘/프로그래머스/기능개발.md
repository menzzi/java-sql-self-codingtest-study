import java.util.*;

class Solution {
    public List<Integer> solution(int[] progresses, int[] speeds) {
        List<Integer> answer = new ArrayList<>();
        Queue<Integer> queue = new LinkedList<>();
        for(int i = 0; i < progresses.length; i++) {
            int count = 100 - progresses[i];
            count /= speeds[i];
            if((100 - progresses[i]) % speeds[i] > 0) count++;
            queue.add(count);
        }
        
        while(!queue.isEmpty()) {
            int cnt = 1;
            int num = queue.poll();
            
            while(!queue.isEmpty() && num >= queue.peek()) {
                queue.poll();
                cnt++;
            }
            answer.add(cnt);
        }
        return answer;
    }
}
