import java.util.*;

class Solution {
    public String[] solution(int[][] line) {
        Set<String> stars = new HashSet<>();
        long minX = Long.MAX_VALUE, maxX = Long.MIN_VALUE;
        long minY = Long.MAX_VALUE, maxY = Long.MIN_VALUE;
        
        for(int i = 0; i < line.length - 1; i++) {
            for(int j = i + 1; j < line.length; j++) {
                long A = line[i][0], B = line[i][1], E = line[i][2];
                long C = line[j][0], D = line[j][1], F = line[j][2];

                
                long one = A * D - B * C;
                long two = B * F - E * D;
                long three = E * C - A * F;

                if(one == 0) continue;
                if(two % one == 0 && three % one == 0) {
                    long x = two / one;
                    long y = three / one;
                    stars.add(x + "," + y);
                    minX = Math.min(minX, x);
                    maxX = Math.max(maxX, x);
                    minY = Math.min(minY, y);
                    maxY = Math.max(maxY, y);
                }
            }
        }
        
        int W = (int) (maxX - minX + 1);
        int H = (int) (maxY - minY + 1);
        
        char[][] board = new char[H][W];
        for(char[] row : board) Arrays.fill(row, '.');
        for(String str : stars) {
            String[] strs = str.split(",");
            long x = Long.parseLong(strs[0]);
            long y = Long.parseLong(strs[1]);
            
            int row = (int)(maxY - y);
            int col = (int)(x - minX);
            board[row][col] = '*';
        }
        
        String[] answer = new String[H];
        StringBuilder sb;
        for(int i = 0; i < H; i++) {
            sb = new StringBuilder();
            for(int j = 0; j < W; j++) {
                sb.append(board[i][j]);
            }
            answer[i] = sb.toString();
        }
        
        return answer;
    }
}
