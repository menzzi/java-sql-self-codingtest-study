```java
class Solution {
    int answer = 0;
    public int solution(int[] numbers, int target) {
        count(numbers, target, 0, 0);
        return answer;
    }
    
    private void count(int[] numbers, int target, int d, int sum) {
        if(d == numbers.length) {
            if (sum == target) answer++;
            return;
        }
        count(numbers, target, d + 1, sum + numbers[d]);
        count(numbers, target, d + 1, sum - numbers[d]);
    }
}
```
