import java.util.*;

class Solution {
    
    private Map<Integer, List<Integer>> graph = new HashMap<>();;
    private Set<Integer> visited = new HashSet<>();
    private int[] answer = new int[2];
    private int a, b;
    
    public int[] solution(int[] nodes, int[][] edges) {
        init(nodes, edges);
        sol(nodes);
        return answer;
    }
    
    private void init(int[] nodes, int[][] edges) {
        for(int n : nodes) {
            graph.put(n, new ArrayList<>());
        }
        
        for(int[] e : edges) {
            graph.get(e[0]).add(e[1]);
            graph.get(e[1]).add(e[0]);
        }
    }
    
    private void sol(int[] nodes) {
        for(int node : nodes) {
            if(visited.contains(node)) continue;
            a = 0;
            b = 0;
            bfs(node);
            
            if(a == 1) {
                answer[1]++;
            }
            if(b == 1) {
                answer[0]++;
            }
        }
    }
    
    private void bfs(int node) {
        Queue<Integer> queue = new LinkedList<>();
        queue.add(node);
        visited.add(node);
        
        while(!queue.isEmpty()) {
            int poll = queue.poll();
            
            int cnt = graph.get(poll).size() - 1;
            
            if(poll % 2 == 0) {
                if(cnt % 2 == 0) {
                    a++;
                } else b++;
            } else {
                if(cnt % 2 == 0) {
                    b++;
                } else a++;
            }
            
            for(int next : graph.get(poll)) {
                if(visited.contains(next)) continue;
                queue.add(next);
                visited.add(next);
            }
        }
    }
}
