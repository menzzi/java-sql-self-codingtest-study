class Solution {
    public int solution(String skill, String[] skill_trees) {
        int answer = skill_trees.length;
        String[] s = skill.split("");
        
        for(String skill_tree : skill_trees) {
            String[] st = skill_tree.split("");
            boolean[] tree = new boolean[s.length];
            boolean flag = true;
            
            for(int i = 0; i < st.length; i++) {
                for(int j = 0; j < s.length; j++) {
                    if(st[i].equals(s[j])) {
                        for(int k = 0; k < j; k++) {
                            if(!tree[k]) {
                                flag = false;
                                answer--;
                                break;
                            }
                        }
                        tree[j] = true;
                    }
                    if(!flag) break;
                }
            }
        }
        return answer;
    }
}
