import java.util.*;

class Solution {
    public int solution(int m, int n, int[][] puddles) {
        int[][] count = new int[n + 1][m + 1];
        
        for(int[] p : puddles) {
            count[p[1]][p[0]] = -1;
        }
        
        count[1][1] = 1;
        
        for(int i = 1; i <= n; i++) {
            for(int j = 1; j <= m; j++) {
                if(count[i][j] == 0) {
                    if(i - 1 > 0 && count[i - 1][j] > 0) {
                        count[i][j] += count[i - 1][j];
                        count[i][j] %= 1000000007;
                    }
                    if(j - 1 > 0 && count[i][j - 1] > 0) {
                        count[i][j] += count[i][j - 1];
                        count[i][j] %= 1000000007;
                    }
                }
            }
        }
        return count[n][m];
    }
}
