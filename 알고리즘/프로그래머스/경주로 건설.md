import java.util.*;

class Solution {
    static final int[] dx = {1, 0, -1, 0};
    static final int[] dy = {0, 1, 0, -1};
    static final int INF = 1_000_000_000;
    
    public int solution(int[][] board) {
        int H = board.length, W = board[0].length;
        if (H == 1 && W == 1) return 0; // 시작=도착

        int[][][] dist = new int[H][W][4];
        for (int y = 0; y < H; y++) {
            for (int x = 0; x < W; x++) {
                Arrays.fill(dist[y][x], INF);
            }
        }
        
        PriorityQueue<int[]> q = new PriorityQueue<>((o1, o2) -> (o1[2] - o2[2]));
        if(board[0][1] == 0) q.add(new int[]{1, 0, 100, 0}); // x, y, cost, dir
        if(board[1][0] == 0) q.add(new int[]{0, 1, 100, 1});
        
        while(!q.isEmpty()) {
            int[] poll = q.poll();
            
            if(poll[2] > dist[poll[1]][poll[0]][poll[3]]) continue;
            
            if(poll[0] == W - 1 && poll[1] == H - 1) {
                return poll[2];
            }
            
            for(int i = 0; i < 4; i++) {
                int nx = poll[0] + dx[i];
                int ny = poll[1] + dy[i];
                
                if(nx < 0 || ny < 0 || nx >= W || ny >= H) continue;
                if(board[ny][nx] == 1) continue;
                int add = 100;
                if(i != poll[3]) add += 500;
                if(poll[2] + add < dist[ny][nx][i]) {
                    dist[ny][nx][i] = poll[2] + add;
                    q.add(new int[]{nx, ny, poll[2] + add, i});
                }
            }
            
        }
        
        return 0;
    }
}
