import java.util.*;

class Solution {
    int[] dx = {1, 0, -1, 0};
    int[] dy = {0, 1, 0, -1};
    public int solution(int[][] maps) {
        int n = maps.length;
        int m = maps[0].length;
        
        Queue<int[]> queue = new LinkedList<>();
        queue.add(new int[] {0, 0, 1});
        boolean[][] visited = new boolean[n][m];
        visited[0][0] = true;
        while(!queue.isEmpty()) {
            int[] poll = queue.poll();
            if(poll[0] == n - 1 && poll[1] == m - 1) {
                return poll[2];
            }
            
            for(int i = 0; i < 4; i++) {
                int nx = poll[0] + dx[i];
                int ny = poll[1] + dy[i];
                
                if(nx < 0 || ny < 0 || nx >= n || ny >= m || maps[nx][ny] == 0 || visited[nx][ny]) continue;
                visited[nx][ny] = true;
                queue.add(new int[]{nx, ny, poll[2] + 1});
            }
            
        }
        return -1;
    }
}
