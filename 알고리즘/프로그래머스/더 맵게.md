import java.util.*;

class Solution {
    public int solution(int[] scoville, int K) {
        int answer = 0;
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        for(int s : scoville) {
            pq.add(s);
        }
        
        while(pq.peek() < K) {
            int newK = pq.poll();
            if(pq.isEmpty()) return -1;
            newK += (pq.poll() * 2);
            answer++;
            pq.add(newK);
        }
        return answer;
    }
}
