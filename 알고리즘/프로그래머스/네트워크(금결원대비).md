import java.util.*;

class Solution {
    boolean[] visited;
    public int solution(int n, int[][] computers) {
        int answer = 0;
        visited = new boolean[n];
        for(int i = 0; i < n; i++) {
            if(!visited[i]) {
                DFS(i, n, computers);
                answer++;
            }
        }
        
        return answer;
    }
    
    private void DFS(int index, int n, int[][] computers) {
        visited[index] = true;
        for(int i = 0; i < n; i++) {
            if(computers[index][i] == 1 && !visited[i]) {
                DFS(i, n, computers);
            }
        }
    }
}
