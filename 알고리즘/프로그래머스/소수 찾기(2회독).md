import java.util.*;

class Solution {
    Set<Integer> set = new HashSet<>();
    public int solution(String numbers) {
        boolean[] visited = new boolean[numbers.length()];
        String[] array = numbers.split("");
        recur(0, visited, array, "");
        
        return set.size();
    }
    
    private void recur(int depth, boolean[] visited, String[] array, String str) {
        if(depth != 0 && depth <= array.length) {
            if(isPrime(str)) {
                System.out.println(str);
                set.add(Integer.parseInt(str));
            }
            if(depth == array.length) return;
        }
        
        for(int i = 0; i < array.length; i++) {
            if(!visited[i]) {
                visited[i] = true;
                recur(depth + 1, visited, array, str + array[i]);
                visited[i] = false;
            }
        }
        
    }
    
    private boolean isPrime(String str) {
        int num = Integer.parseInt(str);
        if(num == 0 || num == 1) return false;
        if(num == 2 || num == 3) return true;
        for(int i = 2; i <= Math.sqrt(num); i++) {
            if(num % i == 0) return false;
        }
        return true;
    }
}
