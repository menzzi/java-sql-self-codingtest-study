import java.util.*;

class Solution {
    public int solution(int[] priorities, int location) {
        int answer = 1;
        PriorityQueue<Integer> pq = new PriorityQueue<>((o1, o2) -> o2 - o1);
        Queue<Integer> waitQueue = new LinkedList<>();
        for(int i = 0; i < priorities.length; i++) {
            pq.add(priorities[i]);
            waitQueue.add(i);
        }
        
        while(!pq.isEmpty()) {
            int poll = pq.poll();
            while(true) {
                int poll2 = waitQueue.poll();
                if(priorities[poll2] == poll) {
                    if(poll2 == location) return answer;
                    answer++;
                    break;
                } else
                    waitQueue.add(poll2);
            }
        }
       
        return answer;
    }
}
