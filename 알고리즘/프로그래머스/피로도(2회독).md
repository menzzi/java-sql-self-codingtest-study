import java.util.*;

class Solution {
    private int answer = 0;
    public int solution(int k, int[][] dungeons) {
        boolean[] visited = new boolean[dungeons.length];
        DFS(dungeons, k, 0, visited);
        return answer;
    }
    private void DFS(int[][] dungeons, int currentK, int depth, boolean[] visited) {
        if(currentK >= 0) {
            answer = Math.max(answer, depth);
        } else if(currentK < 0) return;
        
        if(depth == dungeons.length) return;
        
        for(int i = 0; i < dungeons.length; i++) {
            if(!visited[i] && currentK >= dungeons[i][0] && currentK - dungeons[i][1] >= 0) {
                visited[i] = true;
                DFS(dungeons, currentK - dungeons[i][1], depth + 1, visited);
                visited[i] = false;
            }
        }
    }
}
