import java.util.*;

class Solution {
    public int solution(String[] want, int[] number, String[] discount) {
        int answer = 0;
        Map<String, Integer> map = new HashMap<>();
        for(int i = 0; i < 10; i++) {
            map.put(discount[i], map.getOrDefault(discount[i], 0) + 1);
        }
        if(isCorrect(map, want, number)) answer++;
        int index = 10;
        while(index < discount.length) {
            if(map.get(discount[index - 10]) == 1) {
                map.remove(discount[index - 10]);
            } else {
                map.put(discount[index - 10], map.get(discount[index - 10]) - 1);
            }
            map.put(discount[index], map.getOrDefault(discount[index], 0) + 1);
            if(isCorrect(map, want, number)) answer++;
            index++;
        }
        return answer;
    }
    
    private boolean isCorrect(Map<String, Integer> map, String[] want, int[] number) {
        if(map.size() < want.length) return false;
        for(int i = 0; i < want.length; i++) {
            if(!map.containsKey(want[i])) return false;
            if(map.get(want[i]) != number[i]) return false;
        }
        return true;
    }
}
