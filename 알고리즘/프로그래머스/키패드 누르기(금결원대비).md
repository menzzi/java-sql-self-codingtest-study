import java.util.*;

class Solution {
    public String solution(int[] numbers, String hand) {
        StringBuilder answer = new StringBuilder();
        int[] left = {3, 0};
        int[] right = {3, 2};
        for(int num : numbers) {
            if(num == 1) {
                answer.append("L");
                left[0] = 0;
                left[1] = 0;
            } else if (num == 4) {
                answer.append("L");
                left[0] = 1;
                left[1] = 0;
            } else if (num == 7) {
                answer.append("L");
                left[0] = 2;
                left[1] = 0;
            } else if (num == 3) {
                answer.append("R");
                right[0] = 0;
                right[1] = 2;
            } else if (num == 6) {
                answer.append("R");
                right[0] = 1;
                right[1] = 2;
            } else if (num == 9) {
                answer.append("R");
                right[0] = 2;
                right[1] = 2;
            } else {
                int[] number = new int[2];
                if(num == 2) {
                    number[0] = 0;
                    number[1] = 1;
                } else if(num == 5) {
                    number[0] = 1;
                    number[1] = 1;
                } else if(num == 8) {
                    number[0] = 2;
                    number[1] = 1;
                } else if(num == 0) {
                    number[0] = 3;
                    number[1] = 1;
                }
                if(findDistance(left, number) < findDistance(right, number)){
                    answer.append("L");
                    left = number;
                } else if(findDistance(left, number) > findDistance(right, number)) {
                    answer.append("R");
                    right = number;
                } else {
                    if(hand.equals("left")) {
                        answer.append("L");
                        left = number;
                    } else {
                        answer.append("R");
                        right = number;
                    }
                }
            }
        }
        return answer.toString();
    }
    
    private int findDistance(int[] target, int[] num) {
        return Math.abs(target[0] - num[0]) + Math.abs(target[1] - num[1]);
    }
}
