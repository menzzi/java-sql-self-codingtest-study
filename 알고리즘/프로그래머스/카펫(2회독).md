class Solution {
    public int[] solution(int brown, int yellow) {
        int[] answer = new int[2];
        for(int i = 1; i <= brown / 2; i++) {
            for(int j = 1; j <= i; j++) {
                if(i * j == yellow && (2 * i) + 2 * (j + 2) == brown) {
                    answer[0] = i + 2;
                    answer[1] = j + 2;
                    return answer;
                }
            }
        }
        return answer;
    }
}
