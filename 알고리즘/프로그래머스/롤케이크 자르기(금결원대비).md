import java.util.*;

class Solution {
    public int solution(int[] topping) {
        int answer = 0;
        Set<Integer> set = new HashSet<>();
        for(int i = 0; i < topping.length; i++) {
            set.add(topping[i]);
        }
        int tCount = set.size();
        int index = tCount / 2 - 1;
        Map<Integer, Integer> left = new HashMap<>();
        Map<Integer, Integer> right = new HashMap<>();
        for(int i = 0; i <= index; i++) {
            left.put(topping[i], left.getOrDefault(topping[i], 0) + 1);
        }
        for(int i = index + 1; i < topping.length; i++) {
            right.put(topping[i], right.getOrDefault(topping[i], 0) + 1);
        }
        index++;
        while(index < topping.length) {
            if(left.size() == right.size()) answer++;
            left.put(topping[index], left.getOrDefault(topping[index], 0) + 1);
            if(right.get(topping[index]) == 1) right.remove(topping[index]);
            else right.put(topping[index], right.get(topping[index]) - 1);
            index++;
        }
        return answer;
    }
}
