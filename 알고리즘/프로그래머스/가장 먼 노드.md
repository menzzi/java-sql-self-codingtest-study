import java.util.*;

class Solution {
    public int solution(int n, int[][] edge) {
        List<Integer>[] graph = new ArrayList[n + 1];
        int[] count = new int[n + 1];
        
        for(int i = 0; i <= n; i++) {
            graph[i] = new ArrayList<>();
        }
        
        for(int[] e : edge) {
            graph[e[0]].add(e[1]);
            graph[e[1]].add(e[0]);
        }
        
        Queue<Integer> queue = new LinkedList<>();
        queue.add(1);
        boolean[] visited = new boolean[n + 1];
        visited[1] = true;
        
        while(!queue.isEmpty()) {
            int poll = queue.poll();
            
            for(int num : graph[poll]) {
                if(!visited[num]) {
                    count[num] = count[poll] + 1;
                    queue.add(num);
                    visited[num] = true;
                }
            }
        }
        
        int max = 0;
        int answer = 0;
        for(int i : count) {
            if(max < i) {
                max = i;
                answer = 1;
            } else if(max == i) answer++;
        }
        
        return answer;
    }
}
