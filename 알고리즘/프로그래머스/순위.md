import java.util.*;

class Solution {
    public int solution(int n, int[][] results) {
        int answer = 0;
        int[][] win = new int[n + 1][n + 1];
        
        for(int[] r : results) {
            win[r[0]][r[1]] = 1;
            win[r[1]][r[0]] = -1;
        }
        
        for(int i = 1; i <= n; i++) {
            for(int j = 1; j <= n; j++) {
                for(int k = 1; k <= n; k++) {
                    if(win[i][j] == 1 && win[j][k] == 1) {
                        win[i][k] = 1;
                        win[k][i] = -1;
                    }else if(win[i][j] == -1 && win[j][k] == -1) {
                        win[i][k] = -1;
                        win[k][i] = 1;
                    }
                }
            }
        }
        
        for(int i = 1; i <= n; i++) {
            int cnt = 0;
            for(int j = 1; j <= n; j++) {
                if(win[i][j] != 0) cnt++;
            }
            if(cnt == n - 1) answer++;
        }
        
        return answer;
    }
}
