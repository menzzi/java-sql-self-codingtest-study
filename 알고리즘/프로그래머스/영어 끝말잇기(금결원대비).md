import java.util.*;

class Solution {
    public int[] solution(int n, String[] words) {
        int[] answer = new int[2];
        Set<String> set = new HashSet<>();
        set.add(words[0]);
        for(int i = 1; i < words.length; i++) {
            if(!isPass(set, words[i - 1], words[i])) {
                answer[0] = i % n + 1;
                answer[1] = i / n + 1;
                return answer;
            } 
            set.add(words[i]);
        }

        return answer;
    }
    private boolean isPass(Set<String> set, String preW, String curW) {
        if(set.contains(curW)) return false;
        if (preW.charAt(preW.length() - 1) != curW.charAt(0)) return false;
        return true;
    }
}
