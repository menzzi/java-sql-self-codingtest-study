import java.util.*;

class Solution {
    boolean solution(String s) {
        Stack<Character> st = new Stack<>();
        for(int i = 0; i < s.length(); i++) {
            if(st.isEmpty()) {
                if(s.charAt(i) == ')') return false;
                st.push('(');
            } else {
                if(s.charAt(i) == ')') st.pop();
                else st.push('(');
            }
        }
        
        if(!st.isEmpty()) return false;
        return true;
    }
}
