import java.util.*;

class Solution {
    public int solution(int[][] routes) {
        int answer = 1;
        Arrays.sort(routes, (o1, o2) -> o1[1] - o2[1]);
        
        int now = routes[0][1];
        
        for(int[] r : routes) {
            if(r[0] > now) {
                now = r[1];
                answer++;
            }
        }
        
        return answer;
    }
}
