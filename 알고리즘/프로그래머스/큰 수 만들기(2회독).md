import java.util.*;

class Solution {
    public String solution(String number, int k) {
        int nowK = 0;
        Stack<Character> stack = new Stack<>();
        stack.add(number.charAt(0));
        for(int i = 1; i < number.length(); i++) {
            while(nowK < k && !stack.isEmpty() && stack.peek() < number.charAt(i)) {
                stack.pop();
                nowK++;
            }
            stack.add(number.charAt(i));
        }
        
        while(nowK < k) {
            stack.pop();
            nowK++;
        }
        
        String answer = "";
        while(!stack.isEmpty()) {
            answer += stack.pop();
        }
        
        StringBuffer sb = new StringBuffer(answer);
        sb.reverse();
        return sb.toString();
    }
}
