import java.util.*;

class Solution {
    private int[] parent;
    public int solution(int n, int[][] costs) {
        int totalCost = 0;

        parent = new int[n];
        for(int i = 0; i < n; i++) {
            parent[i] = i;
        }
        
        Arrays.sort(costs, (o1, o2) -> (o1[2] - o2[2]));
        for(int[] m : costs) {
            int fromP = findParent(m[0]);
            int toP = findParent(m[1]);
            if(fromP == toP) continue;
            
            parent[toP] = fromP;
            totalCost += m[2];
        }
        
        return totalCost;
    }
    
    private int findParent(int a) {
        if(parent[a] == a) return parent[a];
        return parent[a] = findParent(parent[a]);
    }
}
