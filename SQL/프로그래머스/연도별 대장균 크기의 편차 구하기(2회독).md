```sql
SELECT YEAR(ED1.DIFFERENTIATION_DATE) AS YEAR, (ED2.MAX_SIZE - ED1.SIZE_OF_COLONY) AS YEAR_DEV, ID
FROM ECOLI_DATA ED1, 
    (SELECT YEAR(DIFFERENTIATION_DATE) AS YEAR, MAX(SIZE_OF_COLONY) AS MAX_SIZE
     FROM ECOLI_DATA 
     GROUP BY YEAR) AS ED2
WHERE YEAR(DIFFERENTIATION_DATE) = ED2.YEAR
ORDER BY YEAR, YEAR_DEV
```
