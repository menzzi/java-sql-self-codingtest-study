```sql
SELECT CC.CAR_ID, CC.CAR_TYPE, FLOOR(30 * DAILY_FEE * (100-DISCOUNT_RATE)/100) AS FEE
FROM CAR_RENTAL_COMPANY_CAR CC 
JOIN CAR_RENTAL_COMPANY_DISCOUNT_PLAN CP ON CC.CAR_TYPE = CP.CAR_TYPE
WHERE (CC.CAR_TYPE = '세단' OR CC.CAR_TYPE = 'SUV')
AND CAR_ID NOT IN (
    SELECT CAR_ID 
    FROM CAR_RENTAL_COMPANY_RENTAL_HISTORY
    WHERE START_DATE <= '2022-11-30' AND END_DATE >= '2022-11-01')
AND DURATION_TYPE = '30일 이상'
GROUP BY CC.CAR_ID, CC.CAR_TYPE
HAVING FEE >= 500000 AND FEE <= 2000000
ORDER BY FEE DESC, CAR_TYPE, CAR_ID DESC
```
