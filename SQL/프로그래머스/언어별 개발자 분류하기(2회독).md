```SQL
WITH FRONTEND AS (
    SELECT SUM(CODE) AS FRONT
    FROM SKILLCODES
    WHERE CATEGORY = 'Front End'
), GRADES AS (
    SELECT 
        CASE WHEN SKILL_CODE & FRONT AND SKILL_CODE & (SELECT CODE FROM SKILLCODES WHERE NAME = 'Python') THEN 'A'
        WHEN SKILL_CODE & (SELECT CODE FROM SKILLCODES WHERE NAME = 'C#') THEN 'B'
        WHEN SKILL_CODE & FRONT THEN 'C'
        ELSE NULL END AS GRADE, ID, EMAIL
    FROM DEVELOPERS, FRONTEND 
)
    

SELECT GRADE, ID, EMAIL
FROM GRADES
WHERE GRADE IS NOT NULL
ORDER BY GRADE, ID
```
