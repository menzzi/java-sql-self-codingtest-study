```SQL
SELECT MONTH(START_DATE) AS MONTH, CAR_ID, COUNT(*) AS RECORDS
FROM CAR_RENTAL_COMPANY_RENTAL_HISTORY C1
WHERE DATE_FORMAT(START_DATE, '%Y-%m') >= '2022-08' AND DATE_FORMAT(START_DATE, '%Y-%m') < '2022-11' 
AND CAR_ID IN (SELECT CAR_ID 
      FROM CAR_RENTAL_COMPANY_RENTAL_HISTORY 
      WHERE DATE_FORMAT(START_DATE, '%Y-%m') >= '2022-08' AND DATE_FORMAT(START_DATE, '%Y-%m') < '2022-11' 
      GROUP BY CAR_ID HAVING COUNT(*) >= 5)
GROUP BY MONTH, CAR_ID
ORDER BY MONTH, CAR_ID DESC
```

[오답] → 해결 방법 : 서브쿼리에서 MONTH를 사용하지 말고 CAR_ID만 가져오는 방식이 안전
```SQL
SELECT MONTH, c2.CAR_ID, COUNT(*) AS RECORDS
FROM CAR_RENTAL_COMPANY_RENTAL_HISTORY C1, (SELECT MONTH(START_DATE) AS MONTH, CAR_ID 
      FROM CAR_RENTAL_COMPANY_RENTAL_HISTORY 
      WHERE DATE_FORMAT(START_DATE, '%Y-%m') >= '2022-08' AND DATE_FORMAT(START_DATE, '%Y-%m') < '2022-11' 
      GROUP BY CAR_ID HAVING COUNT(*) >= 5) AS C2 
WHERE C1.CAR_ID = C2.CAR_ID AND DATE_FORMAT(START_DATE, '%Y-%m') >= '2022-08' AND DATE_FORMAT(START_DATE, '%Y-%m') < '2022-11' 
GROUP BY MONTH, CAR_ID
ORDER BY MONTH, CAR_ID DESC
```
